### modules

```bash
"electron": "^22.0.0",
```

### scripts

```bash
# yarn
$ yarn install

# nuxt dev + electron start
$ yarn dev

# nuxt build (for before packaging)
$ nuxt:build

# build for production
$ yarn pkg # for building environment mac+win
$ yarn pkg:mac
$ yarn pkg:win # not tried.
```
